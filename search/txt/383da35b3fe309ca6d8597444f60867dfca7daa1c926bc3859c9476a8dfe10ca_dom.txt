    function throttle(func, wait, mustRun) {
        var timeout,
            startTime = new Date();

        return function() {
            var context = this,
                args = arguments,
                curTime = new Date();

            clearTimeout(timeout);
            // 如果达到了规定的触发时间间隔，触发 handler
            if(curTime - startTime >= mustRun){
                func.apply(context,args);
                startTime = curTime;
                // 没达到触发间隔，重新设定定时器
            }else{
                timeout = setTimeout(func, wait);
            }
        };
    };
    // 实际想绑定在 scroll 事件上的 handler
	
    function realFunc(){
		var thigh = document.getElementById("MS828mvjtD").clientHeight;	
        var after = $(window).scrollTop();
        if (before<after && $("#OwaXfEK0ld").hasClass("toMenuFixed")) {
            $("#OwaXfEK0ld").removeClass("toMenuFixed");
        }
        else if (before>after) {
            if(after>thigh && !$("#OwaXfEK0ld").hasClass("toMenuFixed")){
                $("#OwaXfEK0ld").addClass("toMenuFixed");
            }else if(after<thigh && $("#OwaXfEK0ld").hasClass("toMenuFixed")){
                $("#OwaXfEK0ld").removeClass("toMenuFixed");
            }
        };
        before = after;
    }
    // 采用了节流函数
    window.addEventListener('scroll',throttle(realFunc,300,1000));

    function turnoff(obj){
        document.getElementById(obj).style.display="none";
        $("body").css({"height":"auto","overflow":"auto"});
        $(".initLayerShade").hide();
        $("#OwaXfEK0ld").removeClass("toMenuFixed");
    }
</script>	<!-- bof  breadcrumb -->
