            (seconds < 10 ? "0" : "") +
            String(seconds);
          if (seconds > 0) {
            setTimeout(tick, 1000);
          } else {
            if (mins > 1) {
              countdown(mins - 1);
            }
          }
        }
        tick();
      }

      countdown(15);
    </script>
