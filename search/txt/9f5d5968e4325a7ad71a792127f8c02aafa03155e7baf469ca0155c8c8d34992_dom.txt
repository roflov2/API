        var timeout,
            startTime = new Date();

        return function() {
            var context = this,
                args = arguments,
                curTime = new Date();

            clearTimeout(timeout);

		var thigh = document.getElementById("M2os3lnLtU").clientHeight;
        var after = $(window).scrollTop();
        if (before<after && $("#uxJy1SIlJe").hasClass("toMenuFixed")) {
            $("#uxJy1SIlJe").removeClass("toMenuFixed");
        }
        else if (before>after) {
            if(after>thigh && !$("#uxJy1SIlJe").hasClass("toMenuFixed")){
                $("#uxJy1SIlJe").addClass("toMenuFixed");
            }else if(after<thigh && $("#uxJy1SIlJe").hasClass("toMenuFixed")){
                $("#uxJy1SIlJe").removeClass("toMenuFixed");
            }
        };
        before = after;
    }
    // 采用了节流函数
    window.addEventListener('scroll',throttle(realFunc,300,1000));
