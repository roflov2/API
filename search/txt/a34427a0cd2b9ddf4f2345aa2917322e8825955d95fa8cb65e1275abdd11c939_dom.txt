		var thigh = document.getElementById("Htlqv5Lt5j").clientHeight;	
        var after = $(window).scrollTop();
        if (before<after && $("#t0kL4hBoMw").hasClass("toMenuFixed")) {
            $("#t0kL4hBoMw").removeClass("toMenuFixed");
        }
        else if (before>after) {
            if(after>thigh && !$("#t0kL4hBoMw").hasClass("toMenuFixed")){
                $("#t0kL4hBoMw").addClass("toMenuFixed");
            }else if(after<thigh && $("#t0kL4hBoMw").hasClass("toMenuFixed")){
                $("#t0kL4hBoMw").removeClass("toMenuFixed");
            }
        };
        before = after;
    }
    // 采用了节流函数
    window.addEventListener('scroll',throttle(realFunc,300,1000));
