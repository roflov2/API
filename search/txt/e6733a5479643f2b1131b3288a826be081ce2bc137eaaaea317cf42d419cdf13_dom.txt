//鼠标滚屏
function _mousewheel($this,delta){
	var scrollTop = $('.main').scrollTop();
	var d = 0;
	if($this.next().length && delta < 0){
		//向下滚动
		$this = $this.next();
		d = $this.offset().top;
	} else if($this.prev().length && delta > 0) {
		//向上滚动
		$this = $this.prev();
		d = $this.offset().top;
	}
	if(d){
		//开始执行滚屏方法
		startAnimate($this);
		$('.main').animate({ scrollTop:scrollTop+d },function(){
			//执行目标滚屏的动画
			activeAnimate($this);
		});		
	}
}

//开始滚屏执行 $this 为即将到达的那一屏

}
